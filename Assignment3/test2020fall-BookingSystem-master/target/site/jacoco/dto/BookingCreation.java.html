package dto;

public class BookingCreation {

    public final String end;

    public BookingCreation(int customerId, int employeeId, String date, String start, String end) {
        this.customerId = customerId;
        this.employeeId = employeeId;
        this.date = date;
        this.start = start;
        this.end = end;
    }
